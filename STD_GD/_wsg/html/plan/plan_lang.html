HTML SCRIPT 
									<!-- 페이지별 스크립트 로드 -->
									<script src="../../../common/js/language/guide_example.js"></script>

									<!-- 다국어 선택 data-lang -->
									<div>
										<button type="button" class="btn" data-lang="ko"><span>한국어</span></button>
										<button type="button" class="btn" data-lang="en"><span>영어</span></button>
									</div>

									<!-- 다국어 마크업 data-lang-name -->
									<div>
										<h1 data-lang-name="sec1_tit1">다국어제목</h1>
										<p data-lang-name="sec1_txt1">다국어제목을 설명합니다.</p>
									</div>
									<div>
										<ul>
											<li data-lang-name="sec2_txt1">오늘은 금요일입니다.</li>
											<li data-lang-name="sec2_txt2">내일은 토요일입니다.</li>
										</ul>
									</div>
								
									/*
										다국어 공통처리
										object = [오브젝트변수]
										다국어.언어.속성 = 다국어[언어변수][속성변수]
									*/
									var lang = {
										arrayHtml : null,
										init : function(customLang){
											this.arrayHtml = customLang;
											this.event();
											this.action('ko');
										},
										event : function(){
											var _this = this;
											$('[data-lang]').on('click', function() {
											  var currentLanguage = $(this).attr('data-lang');
											  _this.action(currentLanguage);
											});
										},
										action : function(currentLanguage){
											var _this = this;
											$('[data-lang-name]').each(function(k) {
												$(this).html(_this.arrayHtml[currentLanguage][$(this).attr('data-lang-name')]);
											});
										}
									}

									// 다국어 페이지별
									var ex_lang = {
										ko : {
											sec1_tit1 : '한국어 sec1-tit1',
											sec1_txt1 : '한국어 sec1-txt1',
											sec2_txt1 : '한국어 sec2-txt1',
											sec2_txt2 : '한국어 sec2-txt2',
										},
										en : {
											sec1_tit1 : 'english sec1-tit1',
											sec1_txt1 : 'english sec1-txt1',
											sec2_txt1 : 'english sec2-txt1',
											sec2_txt2 : 'english sec2-txt2',
										}
									}
									$(function(){
										lang.init(ex_lang);
									})

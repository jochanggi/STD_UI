										// Options, Callback()
									});
// GENERAL
mode: 'horizontal', // 모드
infiniteLoop: true, // 반복
speed: 500, // 이동속도

// PAGER
pager: true, // 페이징노출

// CONTROLS
controls: true, // 다음, 이전 컨트롤러
autoControls: false, // 재생, 정지 컨트롤러
autoControlsCombine: false, // 재생, 정지 토글 컨트롤러

// AUTO
auto: false, // 자동전환
pause: 4000, // 다음재생까지 멈춤시간

// CAROUSEL
minSlides: 1, // 표시 할 슬라이드의 최소 수
maxSlides: 1, // 표시 할 최대 슬라이드 수
moveSlides: 0, // 전환시 이동할 슬라이드 수

// CALLBACKS
onSliderLoad: function(currentIndex){} // 슬라이드 로드됐을 때
onSlideBefore: function($slideElement, oldIndex, newIndex) {} // 슬라이드 동작 시작할 때
onSlideAfter: function($slideElement, oldIndex, newIndex) {} // 슬라이드 동작 끝났을 때
onSlideNext: function($slideElement, oldIndex, newIndex) {} // 다음 슬라이드 동작 시작할 때
onSlidePrev: function($slideElement, oldIndex, newIndex) {} // 이전 슬라이드 동작 시작할 때

// METHODES
sample_bxSlider1.goToSlide(N); // N번째 슬라이드로 이동
sample_bxSlider1.goToNextSlide(); // 다음 슬라이드로 이동
sample_bxSlider1.goToPrevSlide(); // 이전 슬라이드로 이동
sample_bxSlider1.startAuto(); // 자동재생 시작
sample_bxSlider1.stopAuto(); // 자동재생 중지
sample_bxSlider1.getCurrentSlide(); // 현재 슬라이드 반환
sample_bxSlider1.getSlideCount(); // 현재 슬라이드 수를 반환
sample_bxSlider1.destroySlider(); // 슬라이드 호출전 상태로 초기화
